14 /**
15  * A CVODE integrator, parameterized by problem type. The template
16  * parameter should be a class implementing most of the functions
17  * specified in the AbstractProblem class. In particular, the rhsFunc()
18  * and denseJac() functions are required for CVODE to do its job.
19  *
20  * The modifiers may be called any number of times before calling
21  * initIntegration(), but certain parameters both in the integrator
22  * and in the problem itself cannot be changed after initIntegration()
23  * is run.
24  *
25  * The following example shows how this class is typically used:
26  *
27 \code
28 Integrator<PROBLEM_TYPE>* integrator;
29 integrator = new Integrator<PROBLEM_TYPE>(p);
30 
31 p.setFinalTime(23.723);
32 p.setTimestep(0.01);
33 integrator->setPointCount(350);
34 integrator->setCvodeMaxInternalSteps(1e8);
35 // ... other calls to change settings or parameters
36 
37 integrator->initIntegration();
38 while (!integrator->done()) // done() returns false iff t > 23.723
39 {
40  // move forward 0.01 time units
41  // this function will also produce data output
42  // (approximately 350 points as set above)
43  integrator->runStep();
44 }
45 integrator->cleanup(); // free some memory.
46 \endcode
47 
48  *
49  * @see The main() function shows how one might use this template class.
50  */

274 template<typename P>
275 void Integrator<P>::initIntegration()
276 {
277  int retval;
278  int vector_size;
279  N_Vector initial_vector;
280 
281  _cvode_mem = CVodeCreate(_integrationMethod, _iterator);
282 
283 #ifdef DEBUG
284  cout << "Ran CVodeCreate(" << _integrationMethod << ", " << _iterator << ") = " << _cvode_mem << endl;
285 #endif
286  if (_cvode_mem == NULL) {
287  throw logic_error("Integrator<P>::initIntegration(): CVodeCreate() failed.");
288  }
289 
290  retval = CVodeSetFdata(_cvode_mem, (void*) &_problem);
291 
292  if (retval != CV_SUCCESS) {
293  throw logic_error("Integrator<P>::initIntegration(): CVodeSetFdata() failed with status "+ retval);
294  }
295 
296  // Copy the problem's initial conditions vector into our own
297  // vector so we can use the same vector in CVodeMalloc() and
298  // CVode()
299  vector_size = _problem.getVectorSize();
300  initial_vector = _problem.getInitialVector();
301  if (_v == NULL)
302  _v = N_VNew_Serial(vector_size);
303  for (int i = 0; i < vector_size; i++)
304  NV_Ith_S(_v, i) = NV_Ith_S(initial_vector, i);
305 
306 #ifdef DEBUG
307  cout << "Initial conditions vector:" << endl;
308  N_VPrint_Serial(_problem.getInitialVector());
309 #endif
310 
311  // Prototype: CVodeMalloc(handle, function, t_0, initialVector,
312  // relTol, absTolVector)
313  retval = CVodeMalloc(_cvode_mem, P::rhsFunc, _problem.getInitialTime(),
314  _v, CV_SV,
315  _problem.getRelativeTolerance(),
316  _problem.getAbsoluteTolerances());
317 
318 #ifdef DEBUG
319  cout << "Ran CVodeMalloc(...) = " << retval << endl;
320 #endif
321 
322  if (retval != CV_SUCCESS) {
323  throw logic_error("Integrator<P>::initIntegration(): CVodeMalloc() failed with status "+ retval);
324  }
325 
326  retval = CVodeSetErrHandlerFn(
327  _cvode_mem, Integrator<P>::cvodeErrorHandler, (void*) this);
328 
329  if (retval != CV_SUCCESS) {
330  throw logic_error("Integrator<P>::initIntegration(): CVodeSetErrHandlerFn() failed with status "+ retval);
331  }
332 
333  // If we're using the Newton iterator, we also need to use the
334  // Jacobian.
335  if (_iterator == CV_NEWTON) {
336  retval = CVDense(_cvode_mem, _problem.getVectorSize());
337 
338  if (retval != CVDENSE_SUCCESS) {
339  throw logic_error("Integrator<P>::initIntegration(): CVDense() failed with status "+ retval);
340  }
341 
342  retval = CVDenseSetJacFn(_cvode_mem, P::denseJac, (void*) &_problem);
343 
344  if (retval != CVDENSE_SUCCESS) {
345  throw logic_error("Integrator<P>::initIntegration(): CVDenseSetJacFn() failed with status "+ retval);
346  }
347  }
348 
349  if (_numPoints != 0)
350  {
351  // Write the first line of output
352  _problem.recordDataInit();
353 
354  // Record the first data point, because runStep() only does
355  // this after running a step.
356  _problem.recordData(_problem.getInitialTime(),
357  _problem.getInitialVector());
358  _lastRecordedTime = _problem.getInitialTime();
359 
360  // Set how often data points are recorded
361  _recordingDeltaT =
362  (_problem.getFinalTime() - _problem.getInitialTime())
363  / (realtype) _numPoints;
364  }
365  else
366  {
367  _lastRecordedTime = _problem.getInitialTime();
368  _recordingDeltaT = 2 * (_problem.getFinalTime()
369  - _problem.getInitialTime());
370  }
371 
372  // Set the number of internal steps -- high values good!
373  setCvodeMaxInternalSteps();
374 }

376 template<typename P>
377 void Integrator<P>::runStep()
378 {
379  static int retval; // static for speed
380 
381 #ifdef DEBUG2
382  cout << "Running CVode(" << _cvode_mem << ", " << _currentTime + _problem.getTimestep()
383  << ", " << _v << ", " << &_currentTime << ", " << CV_NORMAL << ")" << endl;
384 #endif
385 
386  _problem.preStep(_currentTime, _v);
387 
388  // Advance one timestep, where timestep is defined by the
389  // problem definition.
390  static realtype newTime;
391  newTime = _currentTime + _problem.getTimestep();
392  newTime = newTime > _problem.getFinalTime() ?
393  _problem.getFinalTime() : newTime;
394  retval = CVode(_cvode_mem, newTime, _v, &_currentTime, CV_NORMAL);
395 
396  _problem.postStep(_currentTime, _v);
397 
398  // If it's past time, record a data point
399  if (_shouldUseLogTime)
400  {
401  if (logtime.atNextTimestep(_currentTime))
402  {
403  logtime.increment();
404  _problem.recordData(_currentTime, _v);
405  }
406  }
407  else
408  {
409  if (_currentTime >= _lastRecordedTime + _recordingDeltaT) {
410  _problem.recordData(_currentTime, _v);
411  _lastRecordedTime = _currentTime;
412  }
413  }
414 
415  // If the right-hand function failed,
416  // record the dying data point. The caller can get the error
417  // message and error code from the problem.
418  if (retval == CV_RHSFUNC_FAIL) {
419  if (_currentTime != _lastRecordedTime && _numPoints != 0)
420  _problem.recordData(_currentTime, _v);
421  }
422  if (retval != CV_SUCCESS) {
423  throw logic_error(string("Integrator<P>::runStep(): CVode() failed"));
424  }
425 }
